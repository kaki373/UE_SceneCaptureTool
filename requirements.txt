# UE Scene Capture Tool - Python dependencies

#   "<UE_5.7>/Engine/Binaries/ThirdParty/Python3/Win64/python.exe" -m pip install -r requirements.txt

# Or from inside the editor (Output Log console set to "Python"):
#   import subprocess, sys
#   subprocess.check_call([sys.executable, "-m", "pip", "install", "numpy", "pillow"])

numpy
Pillow

# OpenEXR
# Imath
# opencv-python
